\begin{center}\fcolorbox{red}{white}{

\begin{center}\fcolorbox{red}{white}{
	\( \vv*M{\vv F}(B) = \vv*M{\vv F}(A) + \vv{BA} \wedge \vv F \)
}\end{center}
